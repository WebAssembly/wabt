  (func $get_m (result i32) (i32.load (i32.const 32)))

  (func $g (param i32) (result i32)
    (call $set_m (i32.add (call $get_m) (i32.const 1)))
    (i32.add (get_local 0) (call $get_m)))

    (i32.load8_u (get_local 0)))
  (export "f" $f)
  (export "g" $g)
  (export "h" $h))
